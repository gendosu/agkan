create(input: CreateTaskInput & { status: TaskStatus; created_at: string; updated_at: string }): Task {
  const result = this.db
    .prepare(
      `INSERT INTO tasks (title, body, author, assignees, status, priority, parent_id, branch, created_at, updated_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`
    )
    .run(
      input.title, input.body ?? null, input.author ?? null, input.assignees ?? null,
      input.status, input.priority !== undefined ? input.priority : null,
      input.parent_id !== undefined ? input.parent_id : null,
      input.branch !== undefined ? input.branch : null,  // ← 追加
      input.created_at, input.updated_at
    );
  return this.findById(result.lastInsertRowid as number)!;
}

app.get('/api/git/branches', (c) => {
  try {
    const output = execSync('git branch -a', { cwd: process.cwd() }).toString();
    const branches = output
      .split('\n')
      .map((line) => line.replace(/^\*?\s+/, '').replace(/^remotes\/origin\//, '').trim())
      .filter((line) => line && !line.startsWith('HEAD ->'))
      .filter((line, idx, arr) => arr.indexOf(line) === idx);
    return c.json({ branches });
  } catch {
    return c.json({ branches: [] });
  }
});

if (task.branch) {
  try {
    const branchExists = execSync(`git branch --list ${task.branch}`, { cwd: process.cwd() })
      .toString().trim();
    if (branchExists) {
      execSync(`git checkout ${task.branch}`, { cwd: process.cwd() });
    } else {
      execSync(`git checkout -b ${task.branch}`, { cwd: process.cwd() });
    }
  } catch (e) {
    return c.json({ error: `Failed to checkout branch: ${task.branch}` }, 500);
  }
}

const branchInstruction = !task.branch
  ? `\n\nブランチ指定なし: このタスクのtitleとbodyを読み、作業内容に適したgitブランチ名を生成してください。
形式: task/${taskId}-<kebab-case>（英数字とハイフンのみ、最大60文字）。
生成したブランチ名で git checkout -b を実行してから作業を開始し、
作業開始後に PATCH /api/tasks/${taskId} でbranchフィールドを保存してください。`
  : '';

const prompt = command === 'planning'
  ? `Task ID: ${taskId}\n/agkan-planning-subtask${branchInstruction}`
  : command === 'pr'

let branchSuggestions: string[] = [];
let branchSuggestionsLoaded = false;

async function loadBranchSuggestions(): Promise<void> {
  if (branchSuggestionsLoaded) return;
  try {
    const res = await fetch('/api/git/branches');
    if (res.ok) {
      const data = await res.json() as { branches: string[] };
      branchSuggestions = data.branches;
      branchSuggestionsLoaded = true;
    }
  } catch { /* サジェストなしでも入力は可能 */ }
}

function generateBranchSlug(title: string): string {
  return 'task/' + title
    .toLowerCase().replace(/[^a-z0-9\s-]/g, '').trim()
    .replace(/\s+/g, '-').replace(/-+/g, '-').slice(0, 50);
}

function wireBranchField(taskId: number, taskTitle: string): void {
  const branchInput = document.getElementById('detail-edit-branch') as HTMLInputElement | null;
  const branchDropdown = document.getElementById('detail-branch-dropdown') as HTMLElement | null;
  const autoBtn = document.getElementById('detail-branch-auto-btn') as HTMLButtonElement | null;
  if (!branchInput || !branchDropdown || !autoBtn) return;

  let suggestions: string[] = [], loaded = false;

  async function loadSuggestions(): Promise<void> {
    if (loaded) return;
    try {
      const res = await fetch('/api/git/branches');
      if (res.ok) { ({ branches: suggestions } = await res.json()); loaded = true; }
    } catch { /* サジェストなし */ }
  }

  branchInput.addEventListener('focus', async () => { await loadSuggestions(); renderDropdown(branchInput.value); });
  branchInput.addEventListener('input', () => renderDropdown(branchInput.value));
  branchInput.addEventListener('blur', () => setTimeout(() => { branchDropdown.style.display = 'none'; }, 150));

  autoBtn.addEventListener('click', () => {
    const slug = taskTitle.toLowerCase().replace(/[^a-z0-9\s-]/g, '').trim()
      .replace(/\s+/g, '-').replace(/-+/g, '-').slice(0, 50);
    branchInput.value = `task/${taskId}-${slug}`;
  });
}
